interface TimeEntry {
  // Core fields
  id: string;
  userId: string;
  startTime: Date;
  endTime: Date | null;
  duration: number; // seconds
  
  // Marketing context
  channel: keyof typeof MARKETING_CHANNELS;
  activity: string;
  stage?: string;
  
  // Client/Project hierarchy
  clientId: string;
  projectId?: string;
  campaignId?: string;    // External campaign ID
  accountId?: string;      // Ad account ID
  
  // Billing
  billable: boolean;
  hourlyRate: number;
  rateCardItem?: string;   // Which service from rate card
  poNumber?: string;        // Purchase order ref
  
  // Metadata
  description: string;
  notes?: string;
  tags: string[];          // ["Black Friday", "Q4", "Urgent"]
  attachments?: string[];
  
  // Auto-detection
  autoDetected: boolean;
  autoDetectConfidence?: number;
  detectionSource?: 'url' | 'title' | 'manual';
  
  // QA/Process
  peerReviewed?: boolean;
  checklistCompleted?: boolean;
  checklistId?: string;
}

// Clear metrics for marketers
const MARKETING_METRICS = {
  // Time efficiency
  effectiveHourlyRate: (billings, hours) => billings / hours,
  
  // Cost per managed spend
  timePerThousandSpend: (hours, adSpend) => hours / (adSpend / 1000),
  
  // Blended cost (what client pays vs internal cost)
  blendedCostPerHour: (clientRate, internalCost) => ({
    margin: clientRate - internalCost,
    marginPercent: ((clientRate - internalCost) / clientRate) * 100
  }),
  
  // Service profitability
  serviceProfitability: (revenue, deliveryCost, toolCost) => 
    ((revenue - deliveryCost - toolCost) / revenue) * 100
};
